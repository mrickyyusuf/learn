19 · CSS Functions

Sizing Functions: calc(), min(), max(), clamp()

/* calc() — kalkulasi campuran unit */
.sidebar    { width: calc(300px - 2rem); }
.konten     { width: calc(100% - 300px); }
.gap-trick  { margin-left: calc(-1 * var(--spacing-md)); }
.full-bleed { margin-inline: calc(-1 * var(--padding-halaman)); }

/* min() — ambil nilai TERKECIL dari daftar */
.kartu   { width: min(100%, 400px); }    /* max 400px, tidak overflow container */
.gambar  { width: min(400px, 90vw); }    /* responsif, tidak overflow viewport */

/* max() — ambil nilai TERBESAR */
.konten  { padding: max(1rem, 3vw); }    /* padding minimal 1rem */
.btn     { min-height: max(44px, 3rem); } /* minimal 44px untuk touch target */

/* clamp(minimum, preferred, maximum) — nilai yang dibatasi */
h1    { font-size: clamp(1.5rem, 5vw, 3rem); }  /* responsive typography */
.gap  { gap: clamp(0.5rem, 2vw, 1.5rem); }       /* responsive spacing */
.col  { width: clamp(200px, 30%, 350px); }        /* responsive width dengan batas }

Color Functions

/* rgb() dan rgba() */
color: rgb(46, 125, 50);
color: rgba(46, 125, 50, 0.5);      /* 50% transparan */

/* hsl() dan hsla() — lebih intuitif untuk variasi warna */
--brand: hsl(123, 46%, 34%);
--brand-terang: hsl(123, 46%, 60%); /* sama hue, lightness lebih tinggi */
--brand-gelap:  hsl(123, 46%, 20%); /* sama hue, lightness lebih rendah */

/* oklch() — color space modern yang lebih akurat secara perceptual */
/* Tersedia di Chrome 111+, Firefox 113+, Safari 15.4+ */
color: oklch(50% 0.15 145);  /* lightness 50%, chroma 0.15, hue 145 */

/* color-mix() — campur dua warna */
color: color-mix(in srgb, #2e7d32 70%, white 30%); /* hijau 70% + putih 30% */
background: color-mix(in oklch, var(--brand) 50%, transparent);

Transform Functions

/* Semua nilai transform adalah fungsi */
transform: translateX(10px);
transform: translateY(-20%);
transform: translate(10px, -20%);
transform: scale(1.1);
transform: rotate(45deg);
transform: skewX(15deg);

/* Matrix — gabungan semua transformasi dalam satu fungsi */
/* (jarang ditulis manual, tapi perlu diketahui) */
transform: matrix(a, b, c, d, tx, ty);

/* Gabungan: urutan sangat penting */
/* Dieksekusi dari kanan ke kiri: dulu rotate, lalu translate */
transform: translate(50px, 0) rotate(45deg);

Gradient Functions

/* linear-gradient — gradasi garis lurus */
background: linear-gradient(to right, #1b5e20, #4caf50);
background: linear-gradient(135deg, #c62828, #ff8f00);
background: linear-gradient(to bottom, transparent, rgba(0,0,0,0.7));

/* radial-gradient — gradasi melingkar */
background: radial-gradient(circle at center, #fff9c4, #f57f17);
background: radial-gradient(ellipse at top, #e8f5e9, transparent);

/* conic-gradient — gradasi melingkar/kerucut */
background: conic-gradient(#2e7d32 25%, #c62828 25% 50%, #1565c0 50% 75%, #f9a825 75%);
/* Hasilnya: pie chart 4 bagian */

/* Gradasi sebagai overlay untuk readability teks di atas gambar */
.hero {
  background-image:
    linear-gradient(to bottom, transparent 50%, rgba(0,0,0,0.8) 100%),
    url('gambar.jpg');
  background-size: cover;
  background-position: center;
}

Filter Functions

/* Sudah dibahas di 11-notes.html, ringkasannya: */
filter: blur(4px);                           /* blur */
filter: brightness(0.7);                     /* redup */
filter: contrast(1.5);                       /* kontras tinggi */
filter: grayscale(100%);                     /* hitam putih */
filter: sepia(80%);                          /* efek foto tua */
filter: saturate(1.5);                       /* warna lebih vivid */
filter: hue-rotate(180deg);                  /* putar roda warna */
filter: drop-shadow(2px 4px 6px black);      /* shadow mengikuti bentuk elemen */

/* Kombinasi */
filter: brightness(0.8) contrast(1.2) saturate(1.1);
